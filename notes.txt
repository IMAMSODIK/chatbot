<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use App\Models\Documents;
use Google\Cloud\Vision\V1\ImageAnnotatorClient; // contoh untuk OCR, kalau perlu
use Illuminate\Support\Facades\Storage;
use OpenAI\Client;  // Gemini compatible OpenAI client

class DocumentController extends Controller
{
    protected $openai;

    public function __construct()
    {
        $this->openai = new Client(env('OPENAI_API_KEY')); // Gemini API key
    }

    // Upload & ekstrak teks + buat embedding
    public function store(Request $request)
    {
        $request->validate([
            'files.*' => 'required|mimes:pdf|max:20480',
        ]);

        $results = [];

        foreach ($request->file('files') as $file) {
            $filePath = $file->store('documents', 'public');
            $fileName = pathinfo($file->getClientOriginalName(), PATHINFO_FILENAME);

            // Ekstrak teks dari PDF (gunakan smalot/pdfparser misal)
            $parser = new \Smalot\PdfParser\Parser();
            $pdf = $parser->parseFile(storage_path("app/public/{$filePath}"));
            $extractedText = $pdf->getText();

            // Buat embedding pakai Gemini (OpenAI compatible)
            $embeddingResponse = $this->openai->embeddings()->create([
                'model' => 'gemini-embedding-001',  // sesuaikan nama model Gemini embedding
                'input' => $extractedText,
            ]);

            $embeddingVector = $embeddingResponse->data[0]->embedding;

            // Simpan ke database
            $document = Documents::create([
                'title' => $fileName,
                'file_path' => $filePath,
                'file_type' => 'pdf',
                'extracted_text' => $extractedText,
                'user_id' => auth()->id(),
                'embed' => json_encode($embeddingVector),
            ]);

            $results[] = $document;
        }

        return response()->json([
            'status' => true,
            'data' => $results,
        ]);
    }


    // Cari dokumen berdasarkan pesan user
    public function search(Request $request)
    {
        $request->validate([
            'message' => 'required|string',
        ]);

        $query = $request->message;

        // Buat embedding vektor dari query user
        $embeddingResponse = $this->openai->embeddings()->create([
            'model' => 'gemini-embedding-001',
            'input' => $query,
        ]);
        $queryVector = $embeddingResponse->data[0]->embedding;

        // Ambil dokumen yang punya embedding
        $documents = Documents::whereNotNull('embed')->get();

        // Hitung similarity
        $results = [];
        foreach ($documents as $doc) {
            $docVector = json_decode($doc->embed);

            $similarity = $this->cosineSimilarity($queryVector, $docVector);

            $results[] = [
                'document' => $doc,
                'similarity' => $similarity,
            ];
        }

        // Urutkan descending berdasarkan similarity
        usort($results, fn($a, $b) => $b['similarity'] <=> $a['similarity']);

        // Ambil 3 hasil teratas
        $topResults = array_slice($results, 0, 3);

        // Buat respons string gabungan dari dokumen teratas
        $responseText = "";
        foreach ($topResults as $res) {
            $responseText .= "Menurut dokumen '{$res['document']->title}':\n";
            // Ambil beberapa kalimat awal sebagai ringkasan (misal 200 karakter)
            $snippet = substr($res['document']->extracted_text, 0, 200);
            $responseText .= trim($snippet) . "...\n\n";
        }

        return response()->json([
            'status' => true,
            'response' => $responseText,
            'top_docs' => $topResults,
        ]);
    }


    // Fungsi hitung cosine similarity antara dua vektor
    private function cosineSimilarity(array $vecA, array $vecB): float
    {
        $dot = 0.0;
        $normA = 0.0;
        $normB = 0.0;

        $len = count($vecA);
        for ($i = 0; $i < $len; $i++) {
            $dot += $vecA[$i] * $vecB[$i];
            $normA += $vecA[$i] * $vecA[$i];
            $normB += $vecB[$i] * $vecB[$i];
        }

        if ($normA == 0 || $normB == 0) {
            return 0;
        }

        return $dot / (sqrt($normA) * sqrt($normB));
    }
}

// 1) standards (ISO 20000, ISO 27001)
Schema::create('standards', function (Blueprint $t) {
    $t->id();
    $t->string('code');     // ISO/IEC 20000-1:2018, ISO/IEC 27001:2022
    $t->string('title');
    $t->timestamps();
});

// 2) standard_clauses (tiap pasal/klausul/annex)
Schema::create('standard_clauses', function (Blueprint $t) {
    $t->id();
    $t->foreignId('standard_id')->constrained('standards')->cascadeOnDelete();
    $t->string('clause_code'); // mis: "4.3", "A.5.1"
    $t->text('clause_text');
    $t->longText('embedding')->nullable(); // json
    $t->timestamps();
});

// 3) org_docs (dokumen milik user yang dinilai)
Schema::create('org_docs', function (Blueprint $t) {
    $t->id();
    $t->string('title');
    $t->string('file_path');
    $t->timestamps();
});

// 4) org_doc_chunks (chunk halaman user)
Schema::create('org_doc_chunks', function (Blueprint $t) {
    $t->id();
    $t->foreignId('org_doc_id')->constrained('org_docs')->cascadeOnDelete();
    $t->integer('page_number')->nullable();
    $t->text('chunk_text');
    $t->longText('embedding')->nullable(); // json
    $t->timestamps();
});

KEBIJAKAN PELINDUNGAN DATA PRIBADI

dikumpulkan perusahaan secara langsung maupun tidak langsung melalui sistem elektronik atau

pribadi tersebut yang mencakup pemerolehan dan pengumpulan, pengolahan dan penganalisisan, 
penyimpanan, perbaikan dan pembaruan, penampilan, pengumuman, transfer, penyebarluasan, 
pengungkapan, dan/atau penghapusan atau pemusnahan harus dilakukan sesuai dengan prinsip 
pelindungan data pribadi dan ada dasar pemrosesan data pribadi. 
Sebagai pengendali dan pengelola data pribadi, Peruri memiliki komitmen untuk melindungi data 
pribadi termasuk keamanannya dan melindungi hak-hak pemilik data pribadi serta terus menerus 
meningkatkan kemampuannya dalam mematuhi peraturan perundangan yang berlaku.  
Komitmen pelindungan data pribadi Peruri dituangkan dalam bentuk Kebijakan Pelindungan Data 
Pribadi.  
 
2. Tujuan 
1. Memandu karyawan Peruri dalam melindungi data pribadi karyawan dan non karyawan yang 
mencakup perolehan dan pengumpulan, pengolahan dan penganalisisan, penyimpanan, 
penampilan, pengumuman, pengalihan, penyebarluasan atau pengungkapan data pribadi, 
perbaikan dan pembaruan, serta penghapusan atau pemusnahan data pribadi. 
2. Memastikan agar pengelolaan data pribadi di lingkungan Peruri mematuhi peraturan perundang-
undangan yang berlaku.

Kebebijakan ini berlaku untuk:

atau dapat diidentifikasi secara tersendiri atau dikombinasi dengan informasi lainnya baik secara 
langsung maupun tidak langsung melalui sistem elektronik atau nonelektronik. 
2. Data pribadi yang dilindungi dapat berupa data digital (elektronik) atau non-digital (manual). 
 
4. Definisi 
1. Data pribadi adalah setiap data tentang seseorang baik yang teridentifikasi dan/atau dapat 
diidentifikasi secara tersendiri atau dikombinasi dengan informasi lainnya baik secara langsung 
maupun tidak langsung melalui sistem elektronik dan/atau non-elektronik (secara manual). Data 
Pribadi dapat dikategori menjadi dua yakni Data Pribadi Umum dan Data Pribadi Spesifik. 
2. Data Pribadi Umum merupakan yang mengidentifikasi seseorang secara umum meliputi antara 
lain: 
a. Nama Lengkap 
b. Alamat 
c. Jenis Kelamin 
d. Kewarganegaraan 
e. Agama 
f. Nomor telepon 
g. Email 
h. Data Pribadi yang dikombinasikan untuk mengidentifikasi seseorang seperti data keluarga 
yang bersifat umum

jawab untuk mengembangkan dan menerapkan kebijakan pelindungan data pribadi karyawan dan 
pelanggan dari akses secara tidak sah, serta memastikan bahwa perusahaan persyaratan

13. Risiko dan Dampak Pelindungan Data Pribadi  
1. Risiko dan dampak pemrosesan data pribadi selama siklus hidupnya harus dikaji, diukur, 
didokumentasikan dan ditangani dengan merujuk pada ketentuan manajemen risiko yang berlaku 
di perusahaan. 
2. Penilaian risiko dan dampak pelindungan data pribadi minimum mencakup antara lain: 
 kondisi pelindungan data pribadi saat ini; 
 penggunaan TI dalam pemrosesan data pribadi; 
 perubahan konteks internal dan eksternal yang mempengaruhi proses pelindungan data 
pribadi;  
 tren dan/atau riwayat kasus kebocoran data pribadi, baik di luar atau di dalam perusahaan, 
jika ada; 
 jenis insiden dan/atau kegagalan terhadap pelindungan data pribadi lainnya 
3. Rencana penanganan risiko dan dampak pelindungan data pribadi harus diterapkan dan dipantau 
untuk menjamin agar dampaknya dapat diperkecil pada tingkat yang dapat diterima perusahaan.

14. Kegagalan dan/atau Insiden Pelindungan Data Pribadi  
1. Setiap karyawan yang mengetahui adanya dugaan pelanggaran atau insiden data pribadi harus 
segera melaporkan kepada Data Protection Officer (DPO) untuk dikaji dan diverifikasi validitas 
dan ketepatan ketepatannya. 
2. Insiden dan/atau kegagalan pelindungan data pribadi harus dilaporkan kepada lembaga yang 
berwenang dan kepada pemilik data dalam waktu maksimum 3 x 24 jam setelah Data Protection 
Officer (DPO) menyatakan laporan insiden dan/atau kegagalan pelindungan data pribadi valid dan 
sah. 
3. Insiden dan/atau kegagalan pelindungan data pribadi dalam bentuk sistem elektronik harus 
dikoordinasikan bersama Divisi TI untuk penyelidikan lebih lanjut. 
4. Penanganan insiden dan/atau kegagalan pelindungan data pribadi secara rinci didefinisikan 
secara terpisah dalam Prosedur Penanganan Insiden Data Pribadi. 
 
15. Keamanan Data Pribadi  
Peruri menerapkan langkah-langkah pengamanan teknologi, organisasi dan fisik untuk melindungi 
data pribadi dari kehilangan, penyalahgunaan, penghancuran. modifikasi secara tidak sah, akses atau 
pengungkapan yang tidak sah baik disengaja maupun tidak disengaja. Pengamanan data pribadi yang 
ditetapkan Peruri meliputi antara lain: 
1. Akses terbatas hanya untuk personel yang berkualifikasi dan berwenang.  
2. Data pribadi dalam format tercetak yang berada di meja kerja harus diamankan dan diawasi dari 
kemungkinan dibaca oleh orang yang tidak berwenang. 
3. Lemari yang berisi data pribadi harus selalu dalam keadaan terkunci. 
4. Komputer harus dilindungi dengan kontrol keamanan antara lain: 
a. Akses komputer dengan password yang kuat dan diubah secara berkala; 
b. Menerapkan screen lock berpassword secara otomatis setelah 5 (lima) menit tidak aktif; 
c. Dimatikan saat ditinggalkan tanpa pengawasan; 
d. Dilengkapi dengan software antivirus yang diperbarui secara berkala   
5. Server yang berisi data pribadi harus ditempatkan di lokasi yang aman dan dilengkapi dengan 
sistem pengamanan fisik dan lingkungan yang memadai. 
6. Setiap pelanggaran terhadap kebijakan pelindungan data pribadi harus dilaporkan dan 
ditindaklanjuti sesuai ketentuan perusahaan dan peraturan perundang-undangan yang berlaku. 
Kontrol-kontrol pengamanan data pribadi lainnya merujuk pada Kebijakan Sistem Manajemen 
Pengamanan Informasi yang merupakan bagian dari Kebijakan Tata Kelola TI perusahaan.
